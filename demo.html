janimate

janimate(objs,props[,param][,callback])

objs type:Object/Array/ 只要是dom对象或者elementCollection of elements
props type:Object 例如{width:'100px',height:'100px'}
param type:Object 可选，用于设置animate的一些属性。例如{delay:1000,easing:'easeInQuad',duration:800}//设置动画延迟1s再执行，动画效果easeInQuad,动画时间800ms
callback type:Function 可选，动画执行完的回调函数

最简单的用法

			
	var demo1=document.getElementById('demo1');
	demo1.onclick=function(){
		new janimate(demo1,{left:Math.random().toFixed(1)*1000+'px'}).start();
	}

使用动画效果ease

			
	var demo2=document.getElementById('demo2');
	demo2.onclick=function(){
		new janimate(demo2,{left:Math.random().toFixed(1)*1000+'px'},{ease:"easeInCubic"}).start();
	}

设置动画延迟时间delay

			
	var demo3=document.getElementById('demo3');
	demo3.onclick=function(){
		new janimate(demo3,{width:(50+Math.random().toFixed(1)*1000)+'px'},{ease:"easeInQuad",delay:500}).start();
	}

设置动画执行时间duration

			
	var demo4=document.getElementById('demo4');
	var anim4=new janimate(demo4,{width:'500px',opacity:0.1},{duration:1000});
	demo4.onclick=function(){
		anim4.start();
	}
